English

The Relative Strength Index (RSI) was introduced by J. Welles Wilder in his 1978 book New Concepts in Technical Trading Systems. It is one of the most well-known momentum oscillators in quant finance, measuring the relative strength of recent gains vs. losses over N periods.

The core idea: prices cannot move in one direction indefinitely. When buying power is exhausted (RSI too high), mean reversion tends to follow. When selling pressure is depleted (RSI too low), a bounce typically occurs.

RSI ranges from 0 to 100:

  • RSI > 70: Overbought â€” potential reversal down
  • RSI < 30: Oversold â€” potential bounce up
  • RSI = 50: Equilibrium between bulls and bears

import pandas as pd
import numpy as np
import yfinance as yf
import matplotlib.pyplot as plt

def compute_rsi(series: pd.Series, period: int = 14) -> pd.Series:
    """
    Compute RSI using Wilder's smoothing (equivalent to EMA alpha=1/period).
    """
    delta = series.diff()
    gain = delta.clip(lower=0)
    loss = -delta.clip(upper=0)

    # Wilder's smoothing: use ewm with adjust=False, alpha=1/period
    avg_gain = gain.ewm(alpha=1/period, adjust=False).mean()
    avg_loss = loss.ewm(alpha=1/period, adjust=False).mean()

    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi


def rsi_signal(df: pd.DataFrame,
               period: int = 14,
               overbought: float = 70,
               oversold: float = 30) -> pd.DataFrame:
    """
    Generate mean-reversion trading signals from RSI.
    
    Long signal:  RSI crosses UP through oversold threshold (30)
    Short signal: RSI crosses DOWN through overbought threshold (70)
    """
    df = df.copy()
    df['RSI'] = compute_rsi(df['Close'], period)
    
    # Crossover signals
    df['prev_RSI'] = df['RSI'].shift(1)
    
    # Long: RSI crosses above oversold
    df['signal'] = 0
    df.loc[(df['prev_RSI'] < oversold) & (df['RSI'] >= oversold), 'signal'] = 1
    # Short: RSI crosses below overbought  
    df.loc[(df['prev_RSI'] > overbought) & (df['RSI'] <= overbought), 'signal'] = -1
    
    return df

ConnorsRSI Composite

Larry Connors built on RSI(2) to create ConnorsRSI, averaging three sub-indicators for better signal quality:

  • RSI(3): Short-term price momentum
  • Streak RSI: RSI of the consecutive up/down days count
  • Percent Rank: 2-day ROC ranked over past 100 days

CRSI = (RSI(3) + StreakRSI + PercentRank) / 3

Research shows CRSI < 10 buy / > 90 sell historically outperforms simple RSI on large-cap ETFs.

def streak_rsi(close: pd.Series, period: int = 2) -> pd.Series:
    """Calculate RSI of the streak (consecutive up/down days)."""
    delta = close.diff()
    streak = pd.Series(0.0, index=close.index)
    
    for i in range(1, len(delta)):
        if delta.iloc[i] > 0:
            streak.iloc[i] = max(streak.iloc[i-1], 0) + 1
        elif delta.iloc[i] < 0:
            streak.iloc[i] = min(streak.iloc[i-1], 0) - 1
        else:
            streak.iloc[i] = 0
    
    return compute_rsi(streak, period)


def percent_rank(series: pd.Series, lookback: int = 100) -> pd.Series:
    """Rolling percentile rank of 2-day ROC."""
    roc2 = series.pct_change(2)
    return roc2.rolling(lookback).rank(pct=True) * 100


def connors_rsi(close: pd.Series) -> pd.Series:
    rsi3 = compute_rsi(close, period=3)
    srsi  = streak_rsi(close, period=2)
    prank = percent_rank(close, lookback=100)
    return (rsi3 + srsi + prank) / 3

def rsi_ma_filter_strategy(df, rsi_period=14, ma_period=200,
                            oversold=30, overbought=70):
    """
    RSI + 200-SMA trend filter:
    - Only go LONG when price is above 200-SMA (uptrend)
    - Only go SHORT when price is below 200-SMA (downtrend)
    This avoids catching falling knives in bear markets.
    """
    df = df.copy()
    df['RSI'] = compute_rsi(df['Close'], rsi_period)
    df['MA200'] = df['Close'].rolling(ma_period).mean()
    df['above_ma'] = df['Close'] > df['MA200']
    
    df['signal'] = 0
    # Long only in uptrend
    df.loc[(df['RSI'] < oversold) & df['above_ma'], 'signal'] = 1
    # Short only in downtrend
    df.loc[(df['RSI'] > overbought) & ~df['above_ma'], 'signal'] = -1
    
    return df
